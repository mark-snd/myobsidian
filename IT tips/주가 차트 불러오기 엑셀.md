Sub getStockPrice()
 
Dim exobj As New WinHttp.WinHttpRequest  
Dim tmp, cellNumber As String  
Dim HtmlPath, StockCode, Sp(30) As String  
Dim Rng As Range
 
For i = 1 To 2  
'Set Rng = Rng.Offset(1)   'cellNumber = "C3"  
cellNumber = "C" & i + 2   StockCode = Range(cellNumber)  
'MsgBox StockCode   HtmlPath = "[http://finance.naver.com/item/main.nhn?code=](http://finance.naver.com/item/main.nhn?code=)" & StockCode
 
exobj.Open "GET", HtmlPath  
exobj.Send  
exobj.WaitForResponse   tmp = exobj.ResponseText   Sp(i) = Split(tmp, "현재가")(1)  
Sp(i) = Split(Sp(i), "전일대비")(0)   Range("D" & i + 2) = Sp(i)
 
Next
 
End Sub
